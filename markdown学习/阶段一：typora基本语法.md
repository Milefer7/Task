{
	int a, b, c;

	printf("请输入a，b，c的值：\n");
	scanf("%d %d %d",&a,&b,&c);
	printf ("a为%d b为%d c为%d\n", a, b, c);
	
	int max = a;
	
	if (b >= c && b >= a)
	{
		max = b;
	}
	else if(c > a && c > b) 
	{
		max = c;
	}
	printf("最大的值是%d", max);
	return 0;
}
